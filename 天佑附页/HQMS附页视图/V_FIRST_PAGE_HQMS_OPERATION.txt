CREATE OR REPLACE VIEW V_FIRST_PAGE_HQMS_OPERATION AS
SELECT
O.PATIENT_ID 患者ID,
O.VISIT_ID 住院次,
O.OPERATION_CODE 手术编码,
O.OPERATION_DESC 手术名称,
O.IS_CHANGETIME 是否择期手术,
O.ANESTHESIA_CLASS ASA分级,
O.NNIS_GRADE NNIS分级,
O.EMERGENCY_INDICATOR 是否术前预防用抗菌药,
O.OPERATION_NOTE 清洁手术围术期预防用抗菌药天数,
O.REOPERATION_PLAN 是否为重返手术室再手术,
O.COMLICATION_ANAESTHESIA 麻醉并发症,
O.OPER_FOREIGN_BODY 术中异物遗留,
O.ISCOMLICATIONOFOPERATION 手术并发症,
O.POH 术后出血或血肿,
O.OPER_WOUND_DEHISCENCE 手术伤口裂开,
O.OPER_DEEP_VENOUS_THRO 术后深静脉血栓,
O.OPER_PHYSIOLOGICAL 术后生理或代谢紊乱,
O.OPER_RESPIRATORY_FAILURE 术后呼吸衰竭,
O.OPER_PUL_EMBOLISM 术后肺部栓塞,
O.ISANALGESIA 术后肺部感染,
O.OPER_SEPTICEMIA 术后败血症,
O.STWEED 术后骨折,
O.OPER_HIP_FRACTURE 术后髋关节骨折,
O.ISANALGESIA_ID 术后人工气道意外脱出,
O.OPR_COST_TIME 手术持续时间
FROM OPERATION O;
