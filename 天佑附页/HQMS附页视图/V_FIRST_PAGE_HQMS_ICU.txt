CREATE OR REPLACE VIEW V_FIRST_PAGE_HQMS_ICU AS
SELECT
G.PATIENT_ID 患者ID,
G.VISIT_ID 住院次,
G.WARD_NAME 重症监护室名称,
G.ENTER_TIME 进入时间,
G.EXIT_TIME 离开时间,
G.ARTIFICIAL_AIRWAY_PROLAPSE 是否发生人工气道脱出,
G.UNEXPECTED_RETURN_ICU 是否非预期的重返重症医学科,
G.RETURN_INTERVAL_TIME 重返间隔时间,
G.IS_DEATH ICU死亡,
G.USE_CVC_DAYS 使用中静脉导管日数,
G.ISCENTERVEININFECTION 是否发生血液感染,
G.USE_RESPIRATOR_DAY 使用呼吸机日数,
G.INFEC_RESPIRATOR 是否发生与呼吸机相关肺炎感染,
G.INDWELL_CATHETER_DAYS 使用留置导尿管日数,
G.INFEC_LIEN_CATHETERIZATION 是否发生泌尿系统感染
FROM GRAVE_WARD_DETAIL G;
