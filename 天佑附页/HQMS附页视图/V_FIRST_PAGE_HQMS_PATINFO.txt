CREATE OR REPLACE VIEW V_FIRST_PAGE_HQMS_PATINFO AS
SELECT
P.PATIENT_ID 患者ID,
P.VISIT_ID 住院次,
P1.SPACEINTIME 距上一次住本院时间,
P1.ISSAMEINREASON 住院是否因同一疾病,
P1.ISRETURNOPERATION 重返手术室再手术,
P1.AGAIN_POSTOPER_DEATH 再手术死亡,
P1.DATECHOSEN_OPER 是否择期手术,
P1.PCI 进行PCI,
P1.CABG_WITH_PCI 进行PCI后同一天进行CABG手术,
P1.USE_ANT_DRUG 住院期间是否应用抗菌药物,
P1.ANTIBIOTICS_TIME_HOUR 住院抗菌药物使用持续时间,
P1.USE_ANT_DRUG_COMB 住院抗菌药物联合用药,
P1.I_OPER_ANTIBACTERIALDRUGS I类手术切口预防性应用抗菌药物,
P1.I_OPER_ANTIBACTERIALDRUGS_HOUR 手术抗菌药物使用持续时间,
P1.PRE_ANT_DRUG_COMB 手术抗菌药物联合用药,
P1.ANT_ETI_CHECK 抗菌药物患者病原学检查送检,
P1.CLEAN_OPER_ADM 清洁手术前给药,
P1.HIP_OPER_ADM 髋关节置换术前给药,
P1.KNEE_OPER_ADM 膝关节置换术前给药,
P1.MYOMECTOMY_ADM 子宫肌瘤切除术前给药,
P1.ISFOREIGNMATTER 发生手术过程中异物遗留,
P1.IS_PRESSURE_ULCER 是否有压疮,
P1.PRESSURE_SORES_TYPE 是否住院期间发生压疮,
P1.PRESSURE_STAGE 压疮分期,
P1.IS_FALL_DOWN 院内跌倒或坠床,
P1.TUMBLING_DAMAGE_DEGREE 跌倒或坠床严重程度,
P1.TUMBLING_OTHER_REASON 跌倒或坠床的原因,
P1.ISPNEUMOTHORAX 发生医源性气胸,
P1.ISPUNCTUREORTEAR 发生医源性意外穿刺伤或撕裂伤,
P1.BLOOD_TRANSFUSION 接受输血,
P1.BLOOD_TRAN_REACTION 输血反应,
P1.INFUSION 接受输液,
P1.ISINFUSION_REACT 输液反应,
P1.INFECT_INDICATOR 医院感染,
P1.INFECT_TIMES 医院感染次数,
P1.INFECT_OPERATION 手术相关医院感染,
P1.OPER_PAT_PUL_INFECT 手术患者肺部感染,
P1.IS_LEAVE_AFTEROPER_INDICATOR 手术患者手术部位感染,
P1.IS_BLOOD_DIALYSIS 血液透析,
P1.ISCENTERVEININFECTION 与血液透析相关感染,
P1.ISINCLINICALPATHWAY 是否实施临床路径管理,
P1.ISFINICLINICALPATHWAY 是否完成临床路径,
P1.CP_EXIT_REASON 退出原因,
P1.IS_CP_VARIATION 是否变异,
P1.CP_VARIATION_REASON 变异原因,
P1.ISRESCUE 有无抢救,
P.EMER_TREAT_TIMES 抢救次数,
P.ESC_EMER_TIMES 抢救成功次数,
(SELECT DIAG_CORRESPONDENCE FROM DIAG_COMPARING WHERE PATIENT_ID=P.PATIENT_ID AND VISIT_ID=P.VISIT_ID AND DIAG_COMPARE_GROUP='门诊与出院') 出院与门诊,
(SELECT DIAG_CORRESPONDENCE FROM DIAG_COMPARING WHERE PATIENT_ID=P.PATIENT_ID AND VISIT_ID=P.VISIT_ID AND DIAG_COMPARE_GROUP='入院与出院') 入院与出院,
(SELECT DIAG_CORRESPONDENCE FROM DIAG_COMPARING WHERE PATIENT_ID=P.PATIENT_ID AND VISIT_ID=P.VISIT_ID AND DIAG_COMPARE_GROUP='术前与术后') 术前与术后,
(SELECT DIAG_CORRESPONDENCE FROM DIAG_COMPARING WHERE PATIENT_ID=P.PATIENT_ID AND VISIT_ID=P.VISIT_ID AND DIAG_COMPARE_GROUP='临床与病理') 临床与病理,
(SELECT DIAG_CORRESPONDENCE FROM DIAG_COMPARING WHERE PATIENT_ID=P.PATIENT_ID AND VISIT_ID=P.VISIT_ID AND DIAG_COMPARE_GROUP='放射与病理') 放射与病理,
(SELECT DIAG_CORRESPONDENCE FROM DIAG_COMPARING WHERE PATIENT_ID=P.PATIENT_ID AND VISIT_ID=P.VISIT_ID AND DIAG_COMPARE_GROUP='手术冰冻与石蜡病理') 手术冰冻与石蜡病理,
P1.DISEASE_QUALITY_COST_CONTROL 单病种质量管理与费用控制病种,
P1.IN_QUALITY_CONTROL 纳入质控,
P1.TUMBLING_REASON 未纳入原因,
P1.DIEDURINGOPERATION 是否发生围术期死亡,
P1.DIEAFTEROPERATION 是否发生术后猝死
FROM  PAT_VISIT P,PAT_VISIT1 P1;
